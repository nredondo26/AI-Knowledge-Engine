# Oracle Database

## Modelado

Oracle es un RDBMS corporativo con soporte ACID, MVCC, particionamiento avanzado, objetos, JSON nativo, Oracle Text y Spatial.

```sql
CREATE TABLE users (
    id          NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    uuid        RAW(16) NOT NULL UNIQUE,
    username    VARCHAR2(64) NOT NULL,
    email       VARCHAR2(255),
    status      VARCHAR2(20) CHECK (status IN ('active','inactive','banned')) DEFAULT 'active',
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE logs (
    id        NUMBER GENERATED BY DEFAULT AS IDENTITY,
    level     VARCHAR2(10),
    message   CLOB,
    logged_at DATE
) PARTITION BY RANGE (logged_at) (
    PARTITION p_2024 VALUES LESS THAN (DATE '2025-01-01'),
    PARTITION p_2025 VALUES LESS THAN (DATE '2026-01-01'),
    PARTITION p_future VALUES LESS THAN (MAXVALUE)
);
```

## Consultas

```sql
-- Jerarquía CONNECT BY
SELECT employee_id, LPAD(' ',2*(LEVEL-1))||first_name AS tree
FROM employees START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id
ORDER SIBLINGS BY first_name;

-- Oracle Text
CREATE INDEX idx_posts_text ON posts(title, body) INDEXTYPE IS CTXSYS.CONTEXT;
SELECT id, title, SCORE(1) AS relevance
FROM posts WHERE CONTAINS(body, 'inteligencia AND artificial', 1) > 0;

-- Pivot
SELECT * FROM (
    SELECT TO_CHAR(created_at,'YYYY') year, status, COUNT(*) cnt FROM users GROUP BY year, status
) PIVOT (SUM(cnt) FOR status IN ('active' AS active, 'inactive' AS inactive));
```

## Índices

```sql
CREATE INDEX idx_covering ON posts(user_id, created_at DESC, view_count) TABLESPACE users;
```

## Conexión desde aplicación

```python
import oracledb

pool = oracledb.create_pool(user="app", password="secret", dsn="localhost:1521/knowledge", min=2, max=20)

with pool.acquire() as conn:
    with conn.cursor() as cur:
        cur.var(int)
        cur.execute("""
            INSERT INTO posts (user_id, title, slug, body)
            VALUES (:1, :2, :3, :4) RETURNING id INTO :5
        """, (42, "Título", "titulo", "Cuerpo...", cur.var(int)))
        conn.commit()
```

## Monitoreo

```sql
SELECT sql_id, executions, ROUND(elapsed_time/1000000/executions,2) AS avg_sec
FROM dba_hist_sqlstat WHERE executions > 0
ORDER BY avg_sec DESC FETCH FIRST 10 ROWS ONLY;

SELECT blocking_session, sid, username, wait_class, event
FROM v$session WHERE blocking_session IS NOT NULL;

SELECT segment_name, ROUND(bytes/1024/1024,2) AS size_mb
FROM dba_segments WHERE owner = 'APP' AND segment_type = 'TABLE'
ORDER BY bytes DESC;
```
